34  class Histogram2 implements Histogram {

"this"
)

38  	  public Histogram2(int span) {


39  	    this.counts = new int[span];

41  	  public synchronized void increment(int item) {


42  	    counts[item] = counts[item] + 1;


43  	  }


44  	  public synchronized int getCount(int item) {


45  	    return counts[item];


46  	  }


47  	  public int getSpan() {


48  	    return counts.length;
